package main

import (
	"fmt"
)

func main() {
	var t int
	fmt.Scanf("%d", &t)
	for ; t > 0; t-- {
		var l int
		fmt.Scanf("%d", &l)
		nums := make([]int, 0)
		for i := 0; i < l; i++ {
			var a int
			fmt.Scanf("%d", &a)
			nums = append(nums, a)
		}
		fmt.Println(nums)
		pairs := 0
		if (nums[0]+nums[l-1])%2 == 0 {
			pairs++
		}
		for i := 0; i < l-1; i++ {
			if (nums[i]+nums[i+1])%2 == 0 {
				pairs++
			}
		}
		if pairs%2 == 0 {
			fmt.Println("cc")
		} else {
			fmt.Println("sjy")
		}
	}
	fmt.Println("End")
}
